Build docker image:

docker build -t astrarag-chatbot:latest .

docker run -p 8000:8000 -p 8501:8501 -e GROQ_API_KEY=groq_api_key --name astrarag-chatbot-container-1 astrarag-chatbot:latest

Start docker container
docker run -p 8000:8000 -p 8501:8501 -e GROQ_API_KEY=groq_api_key astrarag-chatbot:latest

Detach mode
docker run -d -p 8000:8000 -p 8501:8501 -e GROQ_API_KEY=groq_api_key astrarag-chatbot:latest
